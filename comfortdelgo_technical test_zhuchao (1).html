lr_scores = []

for train_index, test_index in skf.split(df_X, df_Y):
    train_X, train_Y = (df_X.iloc[train_index]).copy(), (df_Y.iloc[train_index]).copy()
    test_X, test_Y = (df_X.iloc[test_index]).copy(), (df_Y.iloc[test_index]).copy()
    
    
    # Apply one-hot encoder to each column with categorical data
    OH_encoder = OneHotEncoder(handle_unknown='ignore', sparse=False)
    OH_cols_train = pd.DataFrame(OH_encoder.fit_transform(train_X[object_cols]))
    OH_cols_test = pd.DataFrame(OH_encoder.transform(test_X[object_cols]))
    

    # One-hot encoding removed index; put it back
    OH_cols_train.index = train_X.index
    OH_cols_test.index = test_X.index

    # Remove categorical columns (will replace with one-hot encoding)
    num_X_train = train_X.drop(object_cols, axis=1)
    num_X_test = test_X.drop(object_cols, axis=1)
    num_cols = list(num_X_train.columns)

    scaler = preprocessing.StandardScaler().fit(num_X_train)
    num_X_train = pd.DataFrame(scaler.transform(num_X_train))
    num_X_test = pd.DataFrame(scaler.transform(num_X_test))
    num_X_train.index, num_X_train.columns = train_X.index, num_cols
    num_X_test.index, num_X_test.columns = test_X.index, num_cols
    
    # Add one-hot encoded columns to numerical features
    OH_X_train = pd.concat([num_X_train, OH_cols_train], axis=1)
    OH_X_test = pd.concat([num_X_test, OH_cols_test], axis=1)
    

    
    
    lr_clf = LogisticRegression(random_state=42, class_weight = 'balanced').fit(OH_X_train, train_Y)
    lr_scores.append(roc_auc_score(test_Y, lr_clf.predict_proba(OH_X_test)[:, 1]))
    lr_cm.append(confusion_matrix(test_Y, lr_clf.predict(OH_X_test)))
    

    X_train, X_valid, Y_train, Y_valid = train_test_split(OH_X_train, train_Y, \
        train_size=0.75, test_size=0.25, random_state=0, stratify = train_Y)
    
    xgb_clf = XGBClassifier(n_estimators = 1000, learning_rate = 0.05, scale_pos_weight=9, \
                              objective="binary:logistic", random_state=42, verbosity = 0,use_label_encoder =False)
    xgb_clf.fit(X_train, Y_train, early_stopping_rounds=5, 
             eval_set=[(X_valid, Y_valid)], verbose=False)
    xgb_scores.append(roc_auc_score(test_Y, xgb_clf.predict_proba(OH_X_test)[:, 1]))
    xgb_cm.append(confusion_matrix(test_Y, xgb_clf.predict(OH_X_test)))

from math import sin, cos, sqrt, atan2, radians

# approximate radius of earth in km
def latlong_km(lat1, lon1,lat2,lon2):
    R = 6373.0

    lat1 = radians(lat1)
    lon1 = radians(lon1)
    lat2 = radians(lat2)
    lon2 = radians(lon2)

    dlon = lon2 - lon1
    dlat = lat2 - lat1

    a = sin(dlat / 2)**2 + cos(lat1) * cos(lat2) * sin(dlon / 2)**2
    c = 2 * atan2(sqrt(a), sqrt(1 - a))

    distance = R * c
    return distance

def near_school_dis(latlong, data = df_school):
    # input data: dataframe, default use df_school
    # input latlong: string, same data format as in df_train latlong column
    # output: a list, [dis1, dis2, dis3], be the distance to nearest primary/secondary/integrated school
    lat1, lon1 = latlong.split(',')
    lat1, lon1 = float(lat1), float(lon1)
    
    data['dis'] = data.apply(lambda row:latlong_km(lat1, lon1, float(row['latLong'].strip('"').split(',')[0]), \
            float(row['latLong'].strip('"').split(',')[1])), axis = 1)
    distance = data.groupby('category')['dis'].min()
    result = []
    result.append(distance['Primary Schools'])
    result.append(distance['Secondary Schools'])
    result.append(distance['Integrated Schools'])
    
    return result

def train(model,df_X, df_Y, fold = 5):
    #model: sklearn models
    #df_X: feature dataframe
    #df_Y: label dataframe
    #fold: number of stratified CV
    # return: [average_AUC, [list of AUC in kfold runs], [list of confusion matrix in kfold run]]
    skf = StratifiedKFold(n_splits=fold)
    auc = []
    cm = []
    
    object_cols = list(df_X.select_dtypes(include = ['object']).columns)
    
    for train_index, test_index in skf.split(df_X, df_Y):
        train_X, train_Y = (df_X.iloc[train_index]).copy(), (df_Y.iloc[train_index]).copy()
        test_X, test_Y = (df_X.iloc[test_index]).copy(), (df_Y.iloc[test_index]).copy()
        
        # Apply one-hot encoder to each column with categorical data
        OH_encoder = OneHotEncoder(handle_unknown='ignore', sparse=False)
        OH_cols_train = pd.DataFrame(OH_encoder.fit_transform(train_X[object_cols]))
        OH_cols_test = pd.DataFrame(OH_encoder.transform(test_X[object_cols]))
        
        # One-hot encoding removed index; put it back
        OH_cols_train.index = train_X.index
        OH_cols_test.index = test_X.index

        # Remove categorical columns (will replace with one-hot encoding)
        num_X_train = train_X.drop(object_cols, axis=1)
        num_X_test = test_X.drop(object_cols, axis=1)
        num_cols = list(num_X_train.columns)

        scaler = preprocessing.StandardScaler().fit(num_X_train)
        num_X_train = pd.DataFrame(scaler.transform(num_X_train))
        num_X_test = pd.DataFrame(scaler.transform(num_X_test))
        num_X_train.index, num_X_train.columns = train_X.index, num_cols
        num_X_test.index, num_X_test.columns = test_X.index, num_cols
    
        # Add one-hot encoded columns to numerical features
        OH_X_train = pd.concat([num_X_train, OH_cols_train], axis=1)
        OH_X_test = pd.concat([num_X_test, OH_cols_test], axis=1)
        
        
        if type(model) == XGBClassifier: #use validation dataset for early stopping
            X_train, X_valid, Y_train, Y_valid = train_test_split(OH_X_train, train_Y, \
            train_size=0.75, test_size=0.25, random_state=0, stratify = train_Y)
            
            model.fit(X_train, Y_train, early_stopping_rounds=5, eval_set=[(X_valid, Y_valid)], verbose=False)
            auc.append(roc_auc_score(test_Y, model.predict_proba(OH_X_test)[:, 1]))
            cm.append(confusion_matrix(test_Y, model.predict(OH_X_test))) 
        else: # no need for validation set
            model.fit(OH_X_train, train_Y)
            auc.append(roc_auc_score(test_Y, model.predict_proba(OH_X_test)[:, 1]))
            cm.append(confusion_matrix(test_Y, model.predict(OH_X_test)))
    return [np.mean(auc),auc,cm]
